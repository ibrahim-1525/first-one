    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const inputText = document.getElementById('inputText');
            const analyzeButton = document.getElementById('analyzeButton');
            const chatContainer = document.getElementById('chat-container');
            const loadingIndicator = document.getElementById('loadingIndicator');
            const otherShareCheckbox = document.getElementById('other-share');
            const otherPlatformInput = document.getElementById('other-platform-input');
            
            otherShareCheckbox.addEventListener('change', () => {
                if (otherShareCheckbox.checked) {
                    otherPlatformInput.classList.remove('hidden');
                } else {
                    otherPlatformInput.classList.add('hidden');
                }
            });

            const API_URL = "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=";
            const API_KEY = "AIzaSyDkh6hslbAVYizvMTgmQRBd4wCEXr-IwpU";
            
            function addMessage(text, isUser = false) {
                const messageDiv = document.createElement('div');
                messageDiv.className = `flex ${isUser ? 'justify-end' : 'justify-start'}`;
                messageDiv.innerHTML = `<div class="${isUser ? 'chat-bubble' : 'chat-bubble-ai'}">${text}</div>`;
                chatContainer.appendChild(messageDiv);
                chatContainer.scrollTop = chatContainer.scrollHeight;
            }

            async function analyzeText() {
                const text = inputText.value.trim();
                if (!text) {
                    addMessage("Please enter some text to analyze.", false);
                    return;
                }

                const isPublic = document.getElementById('public-share').checked;
                
                const platforms = [];
                if (document.getElementById('facebook-share').checked) platforms.push('Facebook');
                if (document.getElementById('instagram-share').checked) platforms.push('Instagram');
                if (document.getElementById('twitter-share').checked) platforms.push('Twitter (X)');
                if (document.getElementById('tiktok-share').checked) platforms.push('TikTok');
                if (document.getElementById('linkedin-share').checked) platforms.push('LinkedIn');
                if (otherShareCheckbox.checked) {
                    const otherPlatformValue = otherPlatformInput.value.trim();
                    if (otherPlatformValue) {
                        platforms.push(otherPlatformValue);
                    }
                }
                const platformString = platforms.length > 0 ? platforms.join(', ') : 'Not specified';
                
                addMessage(text, true);
                inputText.value = '';
                loadingIndicator.classList.remove('hidden');
                analyzeButton.disabled = true;

                const prompt = `Analyze the privacy risks of the following text shared online.
                Shared Content: "${text}"
                Audience: ${isPublic ? 'Public' : 'Private'}
                Platform: ${platformString}
                
                Based on this, generate a JSON object with the following structure:
                {
                    "riskScore": {
                        "value": "[number from 1 to 10]",
                        "level": "[low, medium, or high]"
                    },
                    "potentialThreats": "[a list of bullet points explaining how an attacker could misuse this data]",
                    "solutions": "[a list of bullet points with specific, actionable solutions to mitigate the risk]"
                }
                
                The risk score value should be a number from 1 to 10. The risk level should be "low" for 1-3, "medium" for 4-7, and "high" for 8-10.
                The threats and solutions should be concise and easy to understand.`;
                
                const payload = {
                    contents: [{ parts: [{ text: prompt }] }],
                    generationConfig: {
                        responseMimeType: "application/json",
                    }
                };
                
                try {
                    const response = await fetch(API_URL + API_KEY, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) {
                        console.error(`HTTP error! status: ${response.status}`);
                        addMessage(`Error: Could not analyze the text. HTTP status: ${response.status}. Please check the console for more details.`, false);
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }
                    const result = await response.json();
                    const analysisData = JSON.parse(result.candidates[0].content.parts[0].text);
                    displayAnalysis(analysisData);
                } catch (error) {
                    console.error('Fetch error:', error);
                    addMessage("Error: An unexpected error occurred. Please check your API key and network connection, then try again.", false);
                } finally {
                    loadingIndicator.classList.add('hidden');
                    analyzeButton.disabled = false;
                }
            }

            function displayAnalysis(data) {
                const scoreClass = `risk-${data.riskScore.level.toLowerCase()}`;
                let scoreText = `<p class="font-bold text-lg mb-2">Risk Score: <span class="${scoreClass}">${data.riskScore.value}/10 (${data.riskScore.level})</span></p>`;
                let threatsText = `<p class="font-bold text-lg mt-4 mb-2">Potential Threats:</p>
                                   <ul class="list-disc list-inside space-y-1">
                                       ${data.potentialThreats.map(threat => `<li>${threat}</li>`).join('')}
                                   </ul>`;
                let solutionsText = `<p class="font-bold text-lg mt-4 mb-2">Solutions:</p>
                                    <ul class="list-disc list-inside space-y-1">
                                        ${data.solutions.map(solution => `<li>${solution}</li>`).join('')}
                                    </ul>`;
                addMessage(scoreText + threatsText + solutionsText, false);
            }

            analyzeButton.addEventListener('click', analyzeText);
            inputText.addEventListener('keypress', (e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    analyzeText();
                }
            });
        });
    </script>
